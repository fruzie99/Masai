1. Count the total number of students

Query

SELECT COUNT(*) AS total_students
FROM students;


Output

total_students
8

2. Find the average age of students

Query

SELECT AVG(age) AS average_age
FROM students;

Output:
average_age
21.75
3. Find the minimum and maximum age

Query

SELECT 
    MIN(age) AS minimum_age,
    MAX(age) AS maximum_age
FROM students;


Output

minimum_age	maximum_age
18	          25
4. Count how many students are enrolled in each course

Query

SELECT  course,
    COUNT(*) AS student_count
FROM students
GROUP BY course;


Output

course	            student_count
Computer Science	   3
Data Science	       2
AI & ML                2
Cyber Security	       1


5. Count students whose age is greater than 20

Query

SELECT COUNT(*) AS students_above_20 FROM students WHERE age > 20;


Output:

students_above_20
5




6. Display course-wise student count

Query

SELECT course,COUNT(*) AS student_count FROM students GROUP BY course;


Output

course	             student_count
Computer Science	     3
Data Science	         2
AI & ML	                 2
Cyber Security	         1



7. Display only courses where student count is greater than 1

Query:

SELECT course,COUNT(*) AS student_count FROM students GROUP BY course HAVING COUNT(*) > 1;


Output

course	student_count
Computer Science	3
Data Science	2
AI & ML	2
8. Display courses sorted by highest student count first


Query

SELECT course,COUNT(*) AS student_count FROM students GROUP BY course ORDER BY student_count DESC;


Output

course	           student_count
Computer Science	 3
Data Science	     2
AI & ML	             2
Cyber Security       1
